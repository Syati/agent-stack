update_arg() {
  arg="$1"; new="$2"
  old=$(grep "^ARG ${arg}=" "$FILE" | cut -d= -f2)
  if [ "$old" != "$new" ] && [ -n "$new" ]; then
    sed -i.bak "s/^ARG ${arg}=.*/ARG ${arg}=${new}/" "$FILE"
    rm -f "$FILE.bak"
    printf '  %s: %s → %s\n' "$arg" "$old" "$new"
  fi
}
